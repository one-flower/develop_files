# 安装 node 依赖

1. npm install typescript
2. npm install ts-node
3. npm i @types/node -D

# 使用 ts 文件

- ts-node filesName 编译 ts 文件
- tsc 编译为 js 文件
- tsc --init 生成 tsconfig.json
